# Form

A collection of form components built on React Hook Form that provides validation, accessibility, and consistent styling. These components work together to create comprehensive, accessible forms with built-in error handling.

<Callout type="info">
  **Zod Integration**: All form examples in this documentation use Zod for schema validation, providing type-safe validation with excellent developer experience. The form components work seamlessly with React Hook Form and Zod resolver for robust form validation.
</Callout>

## Architecture

The form system consists of several components that work together:

- **`Form`** - Provides form context (alias for React Hook Form's `FormProvider`)
- **`FormField`** - Wraps React Hook Form's `Controller` with field context
- **`FormItem`** - Container for individual form fields with spacing
- **`FormLabel`** - Accessible label with error styling
- **`FormControl`** - Wrapper for form inputs with ARIA attributes
- **`FormDescription`** - Helper text for form fields
- **`FormMessage`** - Error and validation messages

### Form with Dynamic Fields

<ComponentPreview
  description="Form with dynamically addable/removable email fields and Zod validation"
  code={`import { useForm, useFieldArray } from "react-hook-form"
import { zodResolver } from "@hookform/resolvers/zod"
import * as z from "zod"
import {
  Form,
  FormControl,
  FormField,
  FormItem,
  FormLabel,
  FormMessage,
  Input,
  Button,
} from "@helgadigitals/vera-ui"

const dynamicFormSchema = z.object({
  name: z.string().min(1, "Name is required"),
  emails: z.array(z.object({
    value: z.string().email("Please enter a valid email address"),
  })).min(1, "At least one email is required"),
})

export default function DynamicFieldsForm() {
  const form = useForm<z.infer<typeof dynamicFormSchema>>({
    resolver: zodResolver(dynamicFormSchema),
    defaultValues: {
      name: "",
      emails: [{ value: "" }],
    },
  })

  const { fields, append, remove } = useFieldArray({
    control: form.control,
    name: "emails",
  })

  function onSubmit(values: z.infer<typeof dynamicFormSchema>) {
    console.log("Form data:", values)
  }

  return (
    <Form {...form}>
      <form onSubmit={form.handleSubmit(onSubmit)} className="space-y-6">
        <FormField
          control={form.control}
          name="name"
          render={({ field }) => (
            <FormItem>
              <FormLabel>Name</FormLabel>
              <FormControl>
                <Input {...field} />
              </FormControl>
              <FormMessage />
            </FormItem>
          )}
        />

        <div className="space-y-4">
          <FormLabel>Email Addresses</FormLabel>
          {fields.map((field, index) => (
            <FormField
              control={form.control}
              key={field.id}
              name={\`emails.\${index}.value\`}
              render={({ field }) => (
                <FormItem>
                  <div className="flex gap-2">
                    <FormControl className="flex-1">
                      <Input
                        {...field}
                        placeholder="email@example.com"
                      />
                    </FormControl>
                    {fields.length > 1 && (
                      <Button
                        type="button"
                        variant="outline"
                        size="sm"
                        onClick={() => remove(index)}
                      >
                        Remove
                      </Button>
                    )}
                  </div>
                  <FormMessage />
                </FormItem>
              )}
            />
          ))}
          
          <Button
            type="button"
            variant="outline"
            size="sm"
            onClick={() => append({ value: "" })}
          >
            Add Email
          </Button>
        </div>

        <Button type="submit">Submit</Button>
      </form>
    </Form>
  )
}`}
>
  <DynamicFieldsFormExample />
</ComponentPreview>

### Async Validation

<ComponentPreview
  description="Form with Zod validation and asynchronous username availability check"
  code={`import { useForm } from "react-hook-form"
import { zodResolver } from "@hookform/resolvers/zod"
import * as z from "zod"
import { useState } from "react"
import {
  Form,
  FormControl,
  FormField,
  FormItem,
  FormLabel,
  FormMessage,
  Input,
  Button,
} from "@helgadigitals/vera-ui"

// Mock API call
const checkUsernameAvailability = async (username: string) => {
  await new Promise(resolve => setTimeout(resolve, 1000))
  return username !== "admin" && username !== "test"
}

const asyncValidationSchema = z.object({
  username: z
    .string()
    .min(1, "Username is required")
    .min(3, "Username must be at least 3 characters"),
})

export default function AsyncValidationForm() {
  const [isChecking, setIsChecking] = useState(false)
  
  const form = useForm<z.infer<typeof asyncValidationSchema>>({
    resolver: zodResolver(asyncValidationSchema),
    defaultValues: {
      username: "",
    },
  })

  const validateUsername = async (value: string) => {
    // First run Zod validation
    const zodResult = asyncValidationSchema.shape.username.safeParse(value)
    if (!zodResult.success) {
      return zodResult.error.errors[0].message
    }
    
    // Then run async validation
    setIsChecking(true)
    try {
      const isAvailable = await checkUsernameAvailability(value)
      return isAvailable || "Username is already taken"
    } finally {
      setIsChecking(false)
    }
  }

  function onSubmit(values: z.infer<typeof asyncValidationSchema>) {
    console.log("Form submitted:", values)
  }

  return (
    <Form {...form}>
      <form onSubmit={form.handleSubmit(onSubmit)} className="space-y-6">
        <FormField
          control={form.control}
          name="username"
          rules={{ validate: validateUsername }}
          render={({ field }) => (
            <FormItem>
              <FormLabel>Username</FormLabel>
              <FormControl>
                <div className="relative">
                  <Input {...field} />
                  {isChecking && (
                    <div className="absolute right-3 top-1/2 -translate-y-1/2">
                      <div className="h-4 w-4 animate-spin rounded-full border-2 border-primary border-t-transparent" />
                    </div>
                  )}
                </div>
              </FormControl>
              <FormMessage />
            </FormItem>
          )}
        />

        <Button 
          type="submit" 
          disabled={isChecking || !form.formState.isValid}
        >
          Submit
        </Button>
      </form>
    </Form>
  )
}`}
>
  <AsyncValidationFormExample />
</ComponentPreview>
